    <script>
        let verificationCode = '';
        let pendingUser = null;

        document.addEventListener('DOMContentLoaded', () => {
            // Code input handling
            const codeInputs = document.querySelectorAll('.code-input');
            codeInputs.forEach((input, index) => {
                input.addEventListener('input', (e) => {
                    if (e.target.value.length === 1 && index < 5) {
                        codeInputs[index + 1].focus();
                    }
                });
                input.addEventListener('keydown', (e) => {
                    if (e.key === 'Backspace' && !e.target.value && index > 0) {
                        codeInputs[index - 1].focus();
                    }
                });
            });

            // Form submission
            document.getElementById('register-form').addEventListener('submit', handleRegister);
        });

        function handleRegister(e) {
            e.preventDefault();
            
            const name = document.getElementById('reg-name').value.trim();
            const email = document.getElementById('reg-email').value.trim();
            const password = document.getElementById('reg-password').value;
            const confirm = document.getElementById('reg-confirm').value;

            // Validation
            if (password !== confirm) {
                showToast('Passwords do not match', 'error');
                return;
            }

            if (password.length < 6) {
                showToast('Password must be at least 6 characters', 'error');
                return;
            }

            // Check if email exists
            const users = JSON.parse(localStorage.getItem('users') || '[]');
            if (users.find(u => u.email === email)) {
                showToast('Email already registered', 'error');
                return;
            }

            // Generate verification code
            verificationCode = Math.floor(100000 + Math.random() * 900000).toString();
            pendingUser = {
                id: Date.now().toString(),
                name,
                email,
                password,
                points: 0,
                verified: false,
                referralCode: generateReferralCode(),
                joinedAt: new Date().toISOString(),
                lastBonusClaim: null,
                referrals: []
            };

            // Show code in toast
            showToast(`Verification code: ${verificationCode}`, 'info', 10000);
            
            // Show modal
            document.getElementById('display-code').textContent = verificationCode;
            document.getElementById('verify-modal').classList.add('active');
            
            // Focus first input
            setTimeout(() => document.querySelector('.code-input').focus(), 100);
        }

        function verifyCode() {
            const inputs = document.querySelectorAll('.code-input');
            const enteredCode = Array.from(inputs).map(i => i.value).join('');
            
            if (enteredCode === verificationCode) {
                // Save user
                const users = JSON.parse(localStorage.getItem('users') || '[]');
                pendingUser.verified = true;
                users.push(pendingUser);
                localStorage.setItem('users', JSON.stringify(users));
                
                // Check for referral
                const urlParams = new URLSearchParams(window.location.search);
                const refCode = urlParams.get('ref');
                if (refCode) {
                    processReferral(refCode, pendingUser.id);
                }
                
                showToast('Account verified successfully!', 'success');
                closeVerifyModal();
                
                setTimeout(() => {
                    window.location.href = 'login.html';
                }, 1500);
            } else {
                showToast('Invalid verification code', 'error');
                inputs.forEach(i => {
                    i.value = '';
                    i.classList.add('error');
                });
                setTimeout(() => inputs[0].focus(), 100);
            }
        }

        function closeVerifyModal() {
            document.getElementById('verify-modal').classList.remove('active');
        }

        function generateReferralCode() {
            return 'RH' + Math.random().toString(36).substring(2, 8).toUpperCase();
        }

        function processReferral(code, newUserId) {
            const users = JSON.parse(localStorage.getItem('users') || '[]');
            const referrer = users.find(u => u.referralCode === code);
            
            if (referrer) {
                referrer.points += 100;
                referrer.referrals.push(newUserId);
                localStorage.setItem('users', JSON.stringify(users));
                showToast(`You were referred by ${referrer.name}!`, 'success');
            }
        }
