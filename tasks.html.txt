    <script>
        const taskLinks = [
            'https://omg10.com/4/10603352',
            'https://omg10.com/4/10603351',
            'https://omg10.com/4/10603350',
            'https://omg10.com/4/10603349',
            'https://omg10.com/4/10603348',
            'https://omg10.com/4/10603347',
            'https://omg10.com/4/10603346',
            'https://omg10.com/4/10603356',
            'https://omg10.com/4/10603375',
            'https://omg10.com/4/10603358'
        ];

        let currentTaskId = null;
        let taskTimer = null;

        document.addEventListener('DOMContentLoaded', () => {
            checkAuth();
            loadTasks();
            updateNavPoints();
        });

        function loadTasks() {
            const user = JSON.parse(localStorage.getItem('currentUser'));
            const completed = JSON.parse(localStorage.getItem('completedTasks') || '[]')
                .filter(t => t.userId === user.id);
            
            const container = document.getElementById('tasks-container');
            const completedCount = completed.length;
            
            document.getElementById('progress-text').textContent = `${completedCount}/10 Completed`;
            document.getElementById('daily-progress').style.width = `${(completedCount / 10) * 100}%`;

            container.innerHTML = taskLinks.map((link, index) => {
                const isCompleted = completed.some(t => t.taskId === index + 1);
                return `
                    <div class="task-card glass ${isCompleted ? 'completed' : ''}" data-task-id="${index + 1}">
                        <div class="task-header">
                            <div class="task-number">#${index + 1}</div>
                            <div class="task-reward">
                                <i class="fas fa-coins"></i>
                                <span>100 pts</span>
                            </div>
                        </div>
                        <div class="task-body">
                            <h3>Quick Task ${index + 1}</h3>
                            <p>Complete this simple online task to earn rewards</p>
                        </div>
                        <div class="task-footer">
                            ${isCompleted ? `
                                <div class="completed-badge">
                                    <i class="fas fa-check-circle"></i>
                                    <span>Completed</span>
                                </div>
                            ` : `
                                <button onclick="startTask(${index + 1}, '${link}')" class="btn-task">
                                    <span>Start Task</span>
                                    <i class="fas fa-external-link-alt"></i>
                                </button>
                            `}
                        </div>
                    </div>
                `;
            }).join('');
        }

        function startTask(taskId, link) {
            const user = JSON.parse(localStorage.getItem('currentUser'));
            const completed = JSON.parse(localStorage.getItem('completedTasks') || '[]');
            
            if (completed.some(t => t.taskId === taskId && t.userId === user.id)) {
                showToast('Task already completed!', 'error');
                return;
            }

            currentTaskId = taskId;
            
            // Open link in new tab
            window.open(link, '_blank');
            
            // Show timer modal
            document.getElementById('task-modal').classList.add('active');
            startCountdown();
        }

        function startCountdown() {
            let seconds = 15;
            const circle = document.getElementById('timer-circle');
            const countEl = document.getElementById('timer-count');
            const circumference = 2 * Math.PI * 45;
            
            circle.style.strokeDasharray = circumference;
            
            taskTimer = setInterval(() => {
                seconds--;
                countEl.textContent = seconds;
                
                const offset = circumference - (seconds / 15) * circumference;
                circle.style.strokeDashoffset = offset;
                
                if (seconds <= 0) {
                    clearInterval(taskTimer);
                    completeTask();
                }
            }, 1000);
        }

        function cancelTask() {
            clearInterval(taskTimer);
            document.getElementById('task-modal').classList.remove('active');
            currentTaskId = null;
        }

        function completeTask() {
            const user = JSON.parse(localStorage.getItem('currentUser'));
            
            // Add points
            const users = JSON.parse(localStorage.getItem('users') || '[]');
            const userIndex = users.findIndex(u => u.id === user.id);
            
            if (userIndex !== -1) {
                users[userIndex].points += 100;
                localStorage.setItem('users', JSON.stringify(users));
                localStorage.setItem('currentUser', JSON.stringify(users[userIndex]));
            }
            
            // Save completed task
            const completed = JSON.parse(localStorage.getItem('completedTasks') || '[]');
            completed.push({
                userId: user.id,
                taskId: currentTaskId,
                completedAt: new Date().toISOString()
            });
            localStorage.setItem('completedTasks', JSON.stringify(completed));
            
            // Close modal
            document.getElementById('task-modal').classList.remove('active');
            
            // Success effects
            showToast('Task completed! +100 points', 'success');
            fireConfetti();
            updateNavPoints();
            
            // Refresh tasks
            setTimeout(() => {
                const card = document.querySelector(`[data-task-id="${currentTaskId}"]`);
                card.classList.add('completed');
                card.querySelector('.task-footer').innerHTML = `
                    <div class="completed-badge">
                        <i class="fas fa-check-circle"></i>
                        <span>Completed</span>
                    </div>
                `;
                loadTasks();
            }, 500);
            
            currentTaskId = null;
        }

        function updateNavPoints() {
            const user = JSON.parse(localStorage.getItem('currentUser'));
            if (user) {
                document.getElementById('nav-points').textContent = user.points;
            }
        }
